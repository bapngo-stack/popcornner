        function showLetters(giftNumber) {

    const container = document.getElementById('game-container');

    const lettersContainer = document.getElementById('letters-container');

    const backButton = document.getElementById('back-button');


    // Ẩn tất cả các hộp quà

    document.querySelectorAll('.gift-box').forEach(box => {

        box.style.display = 'none';

    });


    lettersContainer.style.display = 'flex';

    lettersContainer.innerHTML = '';

    backButton.style.display = 'block';


    const messages = giftMessages[giftNumber];

    

    // Chọn ngẫu nhiên 3 thông điệp từ danh sách

    const randomMessages = getRandomMessages(messages, 3);


    randomMessages.forEach((message, index) => {

        const letter = document.createElement('div');

        letter.classList.add('letter');

        letter.textContent = `Lá thư ${index + 1}`;

        letter.onclick = () => {

            letter.style.display = 'none';

            showMessage(message);

        };

        lettersContainer.appendChild(letter);

    });

}

function getRandomMessages(messages, count) {

    const shuffled = [...messages].sort(() => Math.random() - 0.5); // Trộn ngẫu nhiên mảng

    return shuffled.slice(0, count); // Lấy `count` phần tử đầu tiên

}

        function toggleMute() {

            const backgroundMusic = document.getElementById('background-music');

            const muteButton = document.getElementById('mute-button');


            if (backgroundMusic.muted) {

                backgroundMusic.muted = false;

                backgroundMusic.muted = true;

                muteButton.textContent = "🔇";

            }

        }
